Total Count of Frisked, Searched, and Arrested of Subjects by Race

afs %>% 
   count(race, action) %>% 
  mutate(race = fct_infreq(race, n)) %>% 
  mutate(action = fct_relevel(action, c("Stop", "Frisk")),
         race = fct_relevel(race, c("black", "white-hispanic", "white", "other"))
         ) %>% 
  plot_ly(x = ~action, y = ~n, type = "bar", color = ~ race)%>% 
  layout(
    title = 'Frequency of race for Stop, Frisk, Arrest',
    xaxis = list(
      type = 'category',
      title  = 'Action'), 
    yaxis = list(
      title = 'Frequency',
      range = c(0, 6500)
    ) 
    )
